    for attempt := 0; attempt < 5; attempt++ {
        meta, rev, err := wf.Store.GetMeta(ctx, dagID)

    // steps, it transitions directly to paused. Otherwise it transitions to pausing
    // and the scheduler auto-transitions to paused when the last in-flight step completes.

            meta, rev, err := wf.Store.GetMeta(ctx, dagID)
            if err != nil {
                return err
            }
            if !(&DAGState{Meta: meta}).CanPause() {
                return fmt.Errorf("ebind: cannot pause DAG %s: status is %s: %w", dagID, meta.Status, ErrDAGNotRunning)
            }
            hasInFlight := (&DAGState{Meta: meta, Steps: nil}).HasInFlightSteps()
            // Need to actually check steps if HasInFlightSteps might be true.
            // Load steps to check in-flight status.
            if hasInFlight || true { // always load steps for accurate check
                steps, err := wf.Store.ListSteps(ctx, dagID)
                if err != nil {
                    return err

                state := &DAGState{Meta: meta}

                for _, s := range steps {
                    state.Steps[s.StepID] = s

                if !state.HasInFlightSteps() {
                    // No in-flight steps → skip directly to paused
                    meta.Status = DAGStatusPaused

                    meta.Status = DAGStatusPausing
                }

                }
                return err

        }

    }

    WAIT — the above is wrong. `HasInFlightSteps` requires accessing `s.Steps` (the step map), but when called from Pause, we don't have a full DAGState loaded yet. Let me reconsider.

    The correct approach is to always load steps to check HasInFlightSteps. This matches the Cancel pattern which loads steps via `ListSteps`.

    **Corrected Pause implementation:**

    func Pause(ctx context.Context, wf *Workflow, dagID string) error {
        for attempt := 0; attempt < 5; attempt++ {
            meta, rev, err := wf.Store.GetMeta(ctx, dagID)
            if err != nil {
                return err
            }
            if !(&DAGState{Meta: meta}).CanPause() {
                return fmt.Errorf("ebind: cannot pause DAG %s: status is %s: %w", dagID, meta.Status, ErrDAGNotRunning)
            }
            steps, err := wf.Store.ListSteps(ctx, dagID)
            if err != nil {
                return err
            }
            state := &DAGState{Meta: meta, Steps: make(map[string]StepRecord, len(steps))}
            for _, s := range steps {
                state.Steps[s.StepID] = s
            }
            if state.HasInFlightSteps() {
                meta.Status = DAGStatusPausing
            } else {
                meta.Status = DAGStatusPaused
                meta.PausedAt = time.Now().UTC()
            }
            if err := wf.Store.PutMeta(ctx, dagID, meta, rev); err != nil {

                }
                return err
            }
            return nil
        }

    **Resume function** (per D-21, D-22, D-24, D-25, D-31, D-32):
    
    ```go
    // Resume resumes a paused DAG. Transitions paused→running, then publishes a
    // synthetic EventResumed on the event bus so the scheduler re-evaluates ready steps
    // through its normal serialized path. Uses KV CAS with up to 5 retries.
    func Resume(ctx context.Context, wf *Workflow, dagID string) error {
        for attempt := 0; attempt < 5; attempt++ {
            meta, rev, err := wf.Store.GetMeta(ctx, dagID)
            if err != nil {
                return err
            }
            if !(&DAGState{Meta: meta}).CanResume() {
                return fmt.Errorf("ebind: cannot resume DAG %s: status is %s: %w", dagID, meta.Status, ErrDAGNotPaused)
            }
            meta.Status = DAGStatusRunning
            if err := wf.Store.PutMeta(ctx, dagID, meta, rev); err != nil {
                if errors.Is(err, ErrStaleRevision) {
                    continue
                }
                return err
            }
            // Publish resume event for scheduler to re-evaluate ready steps.
            // Goes through the event bus → scheduler's normal serialized event path (D-24).
            ev := Event{Kind: EventResumed, DAGID: dagID}
            data, marshalErr := MarshalEvent(ev)
            if marshalErr != nil {
                return marshalErr
            }
            if pubErr := wf.Bus.Publish(ctx, EventSubject(ev), data); pubErr != nil {
                return pubErr
            }
            return nil
        }
        return ErrStaleRevision
    }
    ```

    **Key design notes (follow exactly):**
    - D-21: Standalone functions (not methods on Workflow) — consistent with Cancel
    - D-22: Return `error` only — no result types
    - D-23: No options structs — keep it simple
    - D-26/D-27: Wrap errors with fmt.Errorf including dagID + status, use %w for sentinel
    - D-28: Pause checks CanPause() before attempting
    - D-29/D-30: HasInFlightSteps=false → direct to paused; true → pausing
    - D-31: Resume checks CanResume() before attempting
    - D-32: Resume CAS-transitions paused→running, THEN publishes EventResumed
    - D-24: Resume publishes event, does NOT enqueue directly
    - 5-attempt CAS loop consistent with Cancel() — retry on ErrStaleRevision only
    - On 5th stale revision failure: return ErrStaleRevision

    **Doc comments:** Both functions must have complete Go-style doc comments explaining behavior, CAS semantics, and error returns.
